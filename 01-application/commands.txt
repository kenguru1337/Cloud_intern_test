
# Сборка Docker-образа
docker build --build-arg AUTHOR="Aleksey Arkhipov" -t kenguru777/echo-server:latest .   

# Запуск контейнера с передачей переменной окружения AUTHOR
docker run -d --name echo-test -p 8000:8000 -e AUTHOR="Aleksey Arkhipov" kenguru777/echo-server:latest 

# Вход в Docker Hub
docker login

# Публикация образа в Docker Hub
docker push kenguru777/echo-server:latest
